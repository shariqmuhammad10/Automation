    function ProcessSubmitRegister() {
        ShowAjaxMessage("", "");
        var grid = $("#PayrollManagementDataView").data("kendoGrid");
        //
        var selectedRowIds = grid.table.attr("data-AptaFund-KendoGrid-selected");

        if (!OnlyOneItemSelected(selectedRowIds)) {
            ShowOkDialog("Selection", "Please select a single record to perform this operation.", null);
            return;
        }
        if (!checkStatus("New")) {
            ShowOkDialog("Notice", "Register is already submitted.", null);
            return;
        }

        $.ajax({
            url: '/QAAutom/Payroll/Payroll/GetIfSubmittedPaymentRequestsExist',
            cache: false,
            dataType: 'JSON',
            success: function (result) {
                if (result.Exists) {
                    ShowYesNoDialog("Confirm", "There are submitted Payment Requests that can be added to this register. Do you want to cancel submitting this register, so you can go back and add the Payment Requests?", function () {
                        return;
                    }, EmployeeWithNegativeGrossPay(selectedRowIds));
                }
                else {
                    EmployeeWithNegativeGrossPay(selectedRowIds);
                }
            }
        });
    }

    function EmployeeWithNegativeGrossPay(selectedRowIds) {
        var data = { id: selectedRowIds };
        $.ajax({
            url: '/QAAutom/Payroll/Payroll/GetIfEmployeesWithNegativeGrossPayExist',
            cache: false,
            dataType: 'JSON',
            data: data,
            success: function (result) {
                if (result.Exists) {
                    ShowYesNoDialog("Confirm", "The employees listed below have negative gross pay, and will be removed from this register if you continue. Cancel Submit Register?", function () {
                        return;
                    }, function () { PayrollRegisterWithClosedJobs(selectedRowIds) });
                }
                else {
                    PayrollRegisterWithClosedJobs(selectedRowIds);
                }
            }
        });
    }

    function PayrollRegisterWithClosedJobs(selectedRowIds) {
        var data = { id: selectedRowIds };
        ShowYesNoDialog("Confirm", "Are you sure you want to submit the selected register?", function () {
            $.ajax({
                url: '/QAAutom/Payroll/Payroll/GetIfPayrollRegisterWithClosedJobsExists',
                cache: false,
                dataType: 'JSON',
                data: data,
                success: function (result) {
                    if (result.Exists) {
                        $.ajax({
                            cache: false,
                            url: "/QAAutom/Payroll/Payroll/GetPayrollRegisterWithClosedJobsData",
                            data: { id: selectedRowIds },
                            dataType: "json",
                            success: function (result) {
                                if (result != null &amp;&amp; result.Success) {
                                    if (result.dc != null) {
                                        $('#RegisterNumberClosedJobs').text(result.dc.RegisterNumber);
                                        $('#PayrollPeriodName').text(result.dc.PayrollPeriodName);
                                        $('#RegisterDescription').text(result.dc.RegisterDescription);
                                        $('#selectedRowIds').val(selectedRowIds);
                                    }

                                    $('#popupWindowClosedJobs').data("kendoWindow").center().open();
                                    $('#popupWindowClosedJobs').data("kendoWindow").refresh();

                                    var grid = $("#PayrollClosedJobsEmployeesDataView").data("kendoGrid");
                                    ClearGridSelectionAndReload(grid, "/QAAutom/GridConfig/SetGridSelectedRecords");
                                }
                                else {
                                    ShowAjaxMessage(result.Message, "error");
                                }
                            }
                        });
                    }
                    else {

                        SubmitRegister(selectedRowIds);
                    }
                }
            });
        });
    }

    var _queueJob = true;
    function SubmitRegister(selectedRowIds) {
        var data = { id: selectedRowIds, jobQueue: _queueJob };
        $.ajax({
            url: '/QAAutom/Payroll/Payroll/SubmitRegister',
            cache: false,
            dataType: 'JSON',
            data: data,
            success: function (result) {
                if (result.Success) {
                    if (result.HasException)
                        showExceptionpopup(selectedRowIds);
                    else {
                        if (_queueJob) {
                            StartJobQueueNotifications();
                            ShowOkDialog("Submit Register", "Your process has been started and is now in the job queue. You will be notified when it is complete.", null);
                        }
                        else
                            ShowAjaxMessage(result.Message, "success");
                    }
                }
                else {
                    ShowAjaxMessage(result.Message, "error");
                    if (result.HasException)
                        showExceptionpopup(selectedRowIds);
                }
                var grid = $("#PayrollManagementDataView").data("kendoGrid");
                grid.dataSource.read();
            }
        });
    }

    function doStickyButtons() {
        var sticky_button_offset_top = $('#menuhr').offset().top;

        // our function that decides weather the navigation bar should have "fixed" css position or not.
        var sticky_button_navigation = function () {
            var scroll_top = $(window).scrollTop(); // our current vertical position from the top
            var screenwidth = $(window).width();
            var buttonswidth = $('#actionbuttons').width();
            var buttonsheight = $('#actionbuttons').height();

            // if we've scrolled more than the navigation, change its position to fixed to stick to top,
            // otherwise change it back to relative
            if (scroll_top &gt; sticky_button_offset_top) {
                $('#menuhr').css({ 'position': 'fixed', 'top': 0, 'left': 40, 'z-index': 11, 'width': (screenwidth - 80) });

                $('#actionbuttonswrapper').removeAttr('style');
            }
        };

        // run our function on load
        sticky_button_navigation();

        // and run it again every time you scroll
        $(window).scroll(function () {
            sticky_button_navigation();
        });
