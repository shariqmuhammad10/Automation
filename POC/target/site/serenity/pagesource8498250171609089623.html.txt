    function SaveRegister() {

        var PayrollPeriodId = $("#PayrollPeriodId").val();
        var TransactionDate = $("#TransactionDate").val();
        var PayDate = $("#PayDate").val();
        var RegisterType = $("#RegisterType").val();

        var validationMessage = '';

        if (typeof PayrollPeriodId == 'undefined' || PayrollPeriodId == '')
            validationMessage += 'Payroll Period is required.' + '&lt;br /&gt;';

        if (typeof TransactionDate == 'undefined' || TransactionDate == '')
            validationMessage += 'Transaction Date is required.' + '&lt;br /&gt;';

        if (typeof PayDate == 'undefined' || PayDate == '')
            validationMessage += 'Pay Date is required.' + '&lt;br /&gt;';

        if (typeof RegisterType == 'undefined' || RegisterType == '')
            validationMessage += 'Register Type is required.' + '&lt;br /&gt;';

        if (validationMessage != '') {

            ShowAjaxMessage(validationMessage, "error");
            return;
        }

        var form = $("#FormPayrollRegister");

        $.ajax({
            type: "POST",
            async: false,
            cache: false,
            url: "/QAAutom/Payroll/Payroll/Save",
            data: form.serialize(),
            success: function (result) {
                if (result.Success === true) {
                    ShowAjaxMessage("", "");
                    ShowAjaxMessage(result.Message, "success");
                    if (result.NewRegister) {
                        window.location.href = "/QAAutom/Payroll/Payroll/RegisterDetails" + "?id=" + result.PayrollRegisterId + "&amp;addMode=" + result.NewRegister;
                    }
                    else {
                        window.location.href = "/QAAutom/Payroll/Payroll/RegisterDetails" + "?id=" + result.PayrollRegisterId;
                    }
                }
                else {
                    ShowAjaxMessage("", "");
                    ShowAjaxMessage(result.Message, "error");
                }
            },
            error: function (jqXHR, textStatus, errorThrown) {
                console.log(textStatus, errorThrown);
            }
        });
    }

    function btnSave__PayrollRegister() {

        var useSupplementalTaxRateOld = $("#UseSupplementalTaxRateOld").val();
        var useSupplementalTaxRate = $("#UseSupplementalTaxRate").is(':checked');
        if (useSupplementalTaxRate) {
            if (useSupplementalTaxRate.toString().toUpperCase() !== useSupplementalTaxRateOld.toUpperCase()) {
                ShowYesNoCancelDialog("Confirm", "You have chosen this register to use the Supplemental Tax Rate for Federal and State Withholding Tax (if applicable). This will charge tax at the Supplemental rate, and not at the rate of the employees deduction sets.&amp;#160; Are you sure you wish to continue this register with the Supplemental Tax Rate?"
                    , function () {
                        SaveRegister();
                    }
                    , function () {
                        $("#UseSupplementalTaxRate").prop('checked', false);
                        SaveRegister();
                    });
            }
            else {
                SaveRegister();
            }
        }
        else {
            SaveRegister();
        }
    }

    function PayrollCycleChange() {
        //ajax call with period id to get transaction data(period end date) and pay date(from periodpaydate field)

        var pcid = $("#PayrollCycleList").val();

        if (pcid == "") {
            pcid = -1;
        }

        var data = {
            PayrollCycleId: pcid
        };

        $.ajax({
            url: '/QAAutom/Payroll/Payroll/GetPayrollPeriodsByPayrollCycle',

            data: data,
            dataType: 'JSON',
            success: function (result) {
                //debugger;
                if (result != null) {
                    var dropdownlist = $("#PayrollPeriodId").data("kendoDropDownList");
                    dropdownlist.setDataSource(result);
                    dropdownlist.refresh();
                }
            }
        });
    }
