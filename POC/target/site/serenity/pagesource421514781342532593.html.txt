    function registerFINSComboBoxEvents() {
        $('.FINSComboBox').each(function () {
            var kendoCombo = $(this).data("kendoComboBox");
            if (kendoCombo != null) {

                kendoCombo.bind("select", function (e) {
                    if (e.item != null) {
                        var dataItem = this.dataItem(e.item.index());
                        if (dataItem.Value == -999) {
                            e.preventDefault();
                        }
                    }

                }
               );

            }
        });
