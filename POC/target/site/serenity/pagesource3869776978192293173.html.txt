    function VendorInvoiceDetailsSave(addAnother, addNext) {
        ShowAjaxMessage("", "");

        var destinationID = $("#DestinationId").val();
        var rName = $('#RequestorName').val();

        var validationMessage = '';

        var vendorId = $("#OrderingVendorId").data("kendoComboBox").value();
        if (isNaN(vendorId)) { vendorId = _vendorId != -1 ? _vendorId: $("#OrderingVendorId").val() }
        $("#OrderingVendorId").val(vendorId);
        var paymentVendorId = $("#PaymentVendorId").data("kendoComboBox").value();
        if (isNaN(paymentVendorId)) { paymentVendorId = $("#OrderingVendorId").val() }
        $("#PaymentVendorId").val(paymentVendorId);

        var employeeId = $("#EmployeeId").data("kendoComboBox").value();
        var pOId = $("#POId").data("kendoDropDownList").value();
        var invoiceType = $("#InvoiceType").data("kendoComboBox").value();
        var vendorInvoiceNumber = $("#VendorInvoiceNumber").val();

        if (typeof invoiceType == 'undefined' || invoiceType == '') {
            validationMessage += 'Invoice type required' + '&lt;br /&gt;';
        }

        if (vendorInvoiceNumber.length &gt; 0) {
            if ($.trim(vendorInvoiceNumber) == '') {
                validationMessage += 'Invoice Number required' + '&lt;br /&gt;';
            }
        }
        else {
            validationMessage += 'Invoice Number required' + '&lt;br /&gt;';
        }

        switch ($("#InvoiceType").data("kendoComboBox").value()) {
            case "R":
                {
                    if (typeof pOId == 'undefined' || pOId == '') {
                        validationMessage += 'PO Number required' + '&lt;br /&gt;';
                    }

                    break;
                }
            case "D":
                {
                    if (typeof vendorId == 'undefined' || vendorId == '' || vendorId == '-1') {
                        validationMessage += 'Ordering Vendor required' + '&lt;br /&gt;';
                    }
                    break;
                }
            case "C":
                {
                    if (typeof pOId == 'undefined' || pOId == '') {
                        if (typeof vendorId == 'undefined' || vendorId == '' || vendorId == '-1') {
                            validationMessage += 'Ordering Vendor required' + '&lt;br /&gt;';
                        }
                    }
                    break;
                }
            case "E":
                {
                    if (typeof employeeId == 'undefined' || employeeId == '') {
                        validationMessage += 'Employee required' + '&lt;br /&gt;';
                    }
                    break;
                }
            default:
                {
                    break;
                }
        }

        if (validationMessage != '') {

            ShowAjaxMessage(validationMessage, "error");
            return;
        }

        var form = $("#FormVendorInvoice");
        $.ajax({
            type: "POST",
            url: form.attr('action'),
            data: form.serialize(),
            error: function (xhr, status, error) {
            },
            success: function (response) {
                if (response.Success == true) {
                    if (addAnother) {
                        NavigateTo("/QAAutom/AccountsPayable/VendorInvoice/CreateInvoice");
                    }
                    else if (addNext) {
                        NavigateTo("/QAAutom/AccountsPayable/VendorInvoice/Detail" + "/" + $("#NextId").val());
                    }
                    else
                        NavigateTo("/QAAutom/AccountsPayable/VendorInvoice/Detail" + "/" + response.InvoiceId);
                }
                else {
                    ShowAjaxMessage(response.Message, "error");
                }
            }
        });
    }

    function ValidateRegularInvoiceLineItemDeletion()
    {
        var poId = $("#POId").data("kendoDropDownList").value();
        var invoiceType = $("#InvoiceType").data("kendoComboBox").value();
        var grid = $("#VendorInvoiceDetailDataView").data("kendoGrid");
        var selectedRowIds = grid.table.attr("data-AptaFund-KendoGrid-selected");
        var gridData = grid.dataSource.data();
        var Ids = selectedRowIds.split(",");

        if(Ids[0] === "-1" &amp;&amp; Ids.length &gt;= 1)
        {
            Ids = new Array();
            for (var i = 0; i &lt; gridData.length; i++) {
                Ids.push(gridData[i].Id);
            }
        }

        if (typeof invoiceType != 'undefined' || invoiceType != '') {
            if (invoiceType == "R")
            {
                if(poId == undefined || poId == null || poId.length &lt;= 0)
                    return false;
                for (var i = 0; i &lt; gridData.length; i++) {
                    if ($.inArray(gridData[i].Id, Ids) !== -1) {
                        var unitsOfMeasure = gridData[i].UnitsOfMeasure;
                        if(unitsOfMeasure != 'Shipping' &amp;&amp; unitsOfMeasure != 'Tax' &amp;&amp; unitsOfMeasure != 'Discount/Premium')
                            return false;
                    }
                }
            }
        }

        return true;
    }

    function doStickyButtons() {
        var sticky_button_offset_top = $('#menuhr').offset().top;

        // our function that decides weather the navigation bar should have "fixed" css position or not.
        var sticky_button_navigation = function () {
            var scroll_top = $(window).scrollTop(); // our current vertical position from the top
            var screenwidth = $(window).width();
            var buttonswidth = $('#actionbuttons').width();
            var buttonsheight = $('#actionbuttons').height();

            // if we've scrolled more than the navigation, change its position to fixed to stick to top,
            // otherwise change it back to relative
            if (scroll_top &gt; sticky_button_offset_top) {
                $('#menuhr').css({ 'position': 'fixed', 'top': 0, 'left': 40, 'z-index': 11, 'width': (screenwidth - 80) });
                $('#actionbuttons').css({ 'position': 'fixed', 'top': 10, 'right': 48, 'z-index': 12, 'width': buttonswidth });

            } else {
                $('#menuhr').removeAttr('style');
                $('#actionbuttons').removeAttr('style');
                $('#actionbuttonswrapper').removeAttr('style');
            }
        };

        // run our function on load
        sticky_button_navigation();

        // and run it again every time you scroll
        $(window).scroll(function () {
            sticky_button_navigation();
        });
    }
