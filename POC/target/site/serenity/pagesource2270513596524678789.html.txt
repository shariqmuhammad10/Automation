    function RefreshOnSubmitMassRequisition(requisitionId, overbudget, moveNext, key) {
        var operationResultId = GetMassActionId();

        $("#divPRMyRequisitionDataView").data("AptaDataView").ExecuteMassOperation('/QAAutom/Purchasing/MyPurchaseRequisition/SubmitMassRequsition',
                     { requisitionId: requisitionId, overBudget: overbudget, moveNext: moveNext, operationResultId: operationResultId, key: key },
                key, function (result) {
                    if (result.Success == 1) {
                        RefreshOnSubmitMassRequisition(result.InvoiceId, false, false, key);

                    } else if (result.Success == 3) {

                        if (submitForAllYes === true) {
                            RefreshOnSubmitMassRequisition(result.InvoiceId, true, false, key);
                        }
                        else if (submitForAllNo === true) {
                            RefreshOnSubmitMassRequisition(result.InvoiceId, false, true, key);
                        }
                        else {

                            if ('True' == 'True') {
                                result.Message += "&lt;br /&gt;&lt;br /&gt;Do you wish to continue?"
                                ShowYesNoDialogWithCheckBox("My Purchase Requisitions", result.Message, function (e) { RefreshOnSubmitMassRequisition(result.InvoiceId, true, false, key); submitForAllYes = e; }, function (e) { RefreshOnSubmitMassRequisition(result.InvoiceId, false, true, key); submitForAllNo = e; });
                            }
                            else {
                                // create an appropriate error message
                                result.Message += "&lt;br /&gt;&lt;br /&gt;Purchase Request cannot be submitted."
                                ShowOkDialog("My Purchase Requisitions", result.Message, null);
                            }
                        }
                    }
                    else if (result.Success == 4) {

                        //4 indicates process end,
                        submitForAllNo = false;
                        submitForAllYes = false;
                        var grid = $("#PRMyRequisitionDataView").data("kendoGrid");
                        ClearGridSelectionAndReload(grid, "/QAAutom/GridConfig/SetGridSelectedRecords");
                        ShowResultGrid();
                    }
                    else {
                        RefreshOnSubmitMassRequisition(result.InvoiceId, false, false, key);

                    }
                }, "popupwindowMassActionResults", false);
    }

    function doStickyButtons() {
        var sticky_button_offset_top = $('#menuhr').offset().top;

        // our function that decides weather the navigation bar should have "fixed" css position or not.
        var sticky_button_navigation = function () {
            var scroll_top = $(window).scrollTop(); // our current vertical position from the top
            var screenwidth = $(window).width();
            var buttonswidth = $('#actionbuttons').width();
            var buttonsheight = $('#actionbuttons').height();

            // if we've scrolled more than the navigation, change its position to fixed to stick to top,
            // otherwise change it back to relative
            if (scroll_top &gt; sticky_button_offset_top) {
                $('#menuhr').css({ 'position': 'fixed', 'top': 0, 'left': 40, 'z-index': 11, 'width': (screenwidth - 80) });
                $('#actionbuttons').css({ 'position': 'fixed', 'top': 10, 'right': 48, 'z-index': 12, 'width': buttonswidth });
                $('#actionbuttonswrapper').css({ 'height': buttonsheight });
            } else {
                $('#menuhr').removeAttr('style');
                $('#actionbuttons').removeAttr('style');
                $('#actionbuttonswrapper').removeAttr('style');
            }
        };

        // run our function on load
        sticky_button_navigation();

        // and run it again every time you scroll
        $(window).scroll(function () {
            sticky_button_navigation();
        });
    }

    function ReloadRequisitioner(approver) {

            cbRqn.DataParameters.ApproverId = approver;

            var approvaltype = $("#PRType").data("kendoComboBox").value();

        }

    function getLocalDateTime(serverDate) {
        if (serverDate != null &amp;&amp; serverDate != "") {
            var serverDateTime = new Date(serverDate);
            var d = new Date();
            var n = d.getTimezoneOffset();
            var hours = n / 60;
            var minutes = n % 60;
            hours = hours * -1;
            minutes = minutes * -1;
            serverDateTime.setHours(serverDateTime.getHours() + hours);
            serverDateTime.setMinutes(serverDateTime.getMinutes() + minutes);
            var dateClientSide = serverDateTime;
            return kendo.toString(dateClientSide, ('MM/dd/yyyy  hh:mm  tt'));
        }
        return "";
    }
