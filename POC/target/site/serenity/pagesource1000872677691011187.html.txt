    function StarImageClick() {
        showSpinner();
        if (saveInProgress == 1) {
            return;
        }
        saveInProgress = 1;

        var reportName = $("#ReportName").data("kendoDropDownList").text();

    function RunReport_OnClick(e) {

        ShowAjaxMessage("", "");
        ClearDBPreportParameterError();
        kendo.ui.progress($(".maindiv"), true);
        if (e == 'GRD') {
            var suppressSubTotals = $('#SuppressSubTotalsCheckBox').prop('checked') ? true : false;
            var suppressGrouping = $('#SuppressGroupingCheckBox').prop('checked') ? true : false;
            var fitToSinglePage = $('#FittoOnePageWideCheckBox').prop('checked') ? true : false;
            var Showreportparametersonreport = $('#ShowreportparametersonreportCheckBox').prop('checked') ? true : false;

            $('#SuppressSubTotals').val(suppressSubTotals);
            $('#SuppressGrouping').val(suppressGrouping);
            $('#FitToSinglePage').val(fitToSinglePage);
            $('#Showreportparametersonreport').val(Showreportparametersonreport);

            $("#ExportFormat").val(e);
            $.ajax({
                type: "POST",
                async: true,
                url: "/QAAutom/Reports/DBReports/RunReport",
                data: FormDBReportParametersSerialize(),
                success: function (result) {
                    kendo.ui.progress($(".maindiv"), false);
                    if (result != "" &amp;&amp; result.Success === undefined || result.Success === null) {
                        $('#dvReportParameters').html(result);
                        $('#dvReportParameters').show();
                    }
                    else if (result != "" &amp;&amp; result.Success === true) {
                         if (result.AddToJobQueue) {
                             StartJobQueueNotifications();
                             ShowOkDialog("Job Queue", "Your process has been started and is now in the job queue. You will be notified when it is complete.", null);
                             setTimeout(function () {
                                 $("#btnReturntoReportScreen_ReportToGrid").click();
                             },
                                 2000);
                            return;
                        }
                        window.location.href = result.Url;
                    }
                    else if (result != "" &amp;&amp; result.ReportExists === true &amp;&amp; result.DataViewEditedResultId &amp;&amp; result.ReportId) {
                        var finsWindow = FINSWindow.FinsWindowManager.CreateFINSWindowObject("popupwindow__ExportToGridWarning")
                         .SetContent({ type: "POST", url: "/QAAutom/Reports/DBReports/ShowExportToGridWarning?reportId=" + result.ReportId + "&amp;DataViewEditedResultId=" + result.DataViewEditedResultId })
                         .SetIsModal(true)
                         .SetTitle("Generated Data Exists")
                         .SetWidth(515)
                         .SetHeight('50%')
                        finsWindow.CreateWindow(true);
                    }
                },
                error: function (jqXHR, textStatus, errorThrown) {
                    console.log(textStatus, errorThrown);
                    kendo.ui.progress($(".maindiv"), false);
                }
            });
        }
        else if (e == 'CPE') {
            $.ajax({
                type: "POST",
                async: true,
                url: "/QAAutom/Reports/DBReports/GetGridReportUrl",
                data: { reportName: $("#ReportName").data("kendoDropDownList").text() },
                success: function (result) {
                    if (result != "" &amp;&amp; result.Success === true) {
                        window.location.href = result.Url;
                    }
                    else if (result != "" &amp;&amp; result.Success === false) {
                        kendo.ui.progress($(".maindiv"), false);

                    }
                },
                error: function (jqXHR, textStatus, errorThrown) {
                    console.log(textStatus, errorThrown);
                    kendo.ui.progress($(".maindiv"), false);
                }
            });
        }
        else {
            //Add check here to check that the selected display setting
            //Then we will create instance of AptaDataView and call export method by setting cycle id
            var IsGridDisplaySetting = $("#IsGridDisplaySetting").val();
            //This check is added for the Grid Display Settings

                        }
                    });
                }

                var suppressGrouping = $('#SuppressGroupingCheckBox').prop('checked') ? true : false;
                var fitToSinglePage = $('#FittoOnePageWideCheckBox').prop('checked') ? true : false;

                $('#FitToSinglePage').val(fitToSinglePage);

    function CreateDataView(result) {
        var dataview = AptaFundDataView.DataViewManager.CreateDataView(result.DataViewId)
                     .SetBaseURL("/QAAutom/controller/action")
                     .SetDataViewParameters(result.optionalParameter == null ? null : result.optionalParameter);

        // divDataViewContainer is used as a placeholder for the DataView div.  It should contain 1 dynamically created element (and only 1 element).
        var newdiv = document.createElement("div");
        newdiv.id = "div" + result.DataViewId;
        $("#divDataViewContainer").empty();
        $("#divDataViewContainer").append(newdiv);
        $("#div" + result.DataViewId).data("AptaDataView", dataview);
    }

    function GetDataView() {
        if ($("#divDataViewContainer") &amp;&amp;
            $("#divDataViewContainer").children() &amp;&amp;
            $("#divDataViewContainer").children().first())
        {
            return $("#divDataViewContainer").children().first().data("AptaDataView");
        }
        return null;
    }

    function SaveReportFilter_OnClick(e) {
        var IsGridDisplaySetting = $("#IsGridDisplaySetting").val();
        ShowAjaxMessage("", "");
        var filterName = $('#txtFilterName').val();

        if (IsGridDisplaySetting != null || IsGridDisplaySetting !== undefined || IsGridDisplaySetting == true) {

            var dataview = GetDataView();
            if (dataview != null) {
                var reportNameValue = GetReportNameValue();
                $.ajax({
                    type: "POST",
                    async: true,
                    url: "/QAAutom/GridConfig/ApplyGirdFilterForReport",
                    data: {
                        gridName: dataview._dataViewName,
                        reportNameValue: reportNameValue,
                        filterName: filterName
                    },
                    success: function (result) {
                        $("#popupWindowReportFilterSaveAs").data("kendoWindow").destroy();
                        LoadReportFilters(filterName);
                        FilterName_OnChange();
                        ShowAjaxMessage('Record saved successfully.', "success");
                    },
                    error: function (jqXHR, textStatus, errorThrown) {
                        console.log(textStatus, errorThrown);
                        kendo.ui.progress($(".maindiv"), false);
                    }
                });
            }
        }
        else {

            if (e == 1) {
                $.ajax({
                    cache: false,
                    url: "/QAAutom/Reports/DBReports/NamedFilterExists",
                    data: { filterName: filterName, reportName: $("#ReportName").data("kendoDropDownList").text() },
                    dataType: "json",
                    success: function (result) {
                        if (result != null &amp;&amp; result.Success) {
                            SaveNameFilter(filterName);
                        }
                        else {
                            ShowYesNoDialog('Override Report Filter', result.Message, function () { SaveNameFilter(filterName); });
                        }
                    }
                });
            }
            else {
                SaveNameFilter(filterName);
            }
        }
    }

    function SaveNameFilter(filterName) {
        var suppressSubTotals = $('#SuppressSubTotalsCheckBox').prop('checked') ? true : false;
        var suppressGrouping = $('#SuppressGroupingCheckBox').prop('checked') ? true : false;

        $('#SuppressSubTotals').val(suppressSubTotals);
        $('#SuppressGrouping').val(suppressGrouping);
        $('#FitToSinglePage').val(fitToSinglePage);
        $('#Showreportparametersonreport').val(Showreportparametersonreport);

        $.ajax({
            type: "POST",
            async: false,
            url: "/QAAutom/Reports/DBReports/SaveNamedFilter",
            data: FormDBReportParametersSerialize(),
            success: function (result) {
                if (result != "" &amp;&amp; result.Success == true) {

                    FilterName_OnChange();

            },
            error: function (jqXHR, textStatus, errorThrown) {
                console.log(textStatus, errorThrown);
                $("#divDBReportsFilterSaveAsAjaxError").html(result.Message);
                $("#divDBReportsFilterSaveAsAjaxError").addClass("global_message error");
            }
        });
    }

    function FormDBReportParametersSerialize() {
        var data = $("#FormDBReportParameters").serializeArray();
        $("#FormDBReportParameters").find("[data-role='multiselect'][data-ismultiselectorder]").each(function () {
            var selected = $(this).data('kendoMultiSelect').value();
            var fromIndex = 0;
            for (i = 0; i &lt; data.length; i++) {
                if (data[i].name == this.name) {
                    data[i].value = selected[fromIndex];
                    fromIndex++;
                }
            }
        });

        return data;
    }

    function doStickyButtons() {
        var sticky_button_offset_top = $('#menuhr').offset().top;

        // our function that decides weather the navigation bar should have "fixed" css position or not.
        var sticky_button_navigation = function () {
            var scroll_top = $(window).scrollTop(); // our current vertical position from the top
            var screenwidth = $(window).width();
            var buttonswidth = $('#actionbuttons').width();
            var buttonsheight = $('#actionbuttons').height();

            // if we've scrolled more than the navigation, change its position to fixed to stick to top,
            // otherwise change it back to relative
            if (scroll_top &gt; sticky_button_offset_top) {
                $('#menuhr').css({ 'position': 'fixed', 'top': 0, 'left': 40, 'z-index': 11, 'width': (screenwidth - 80) });
                $('#actionbuttons').css({ 'position': 'fixed', 'top': 10, 'right': 48, 'z-index': 12, 'width': buttonswidth });
                $('#actionbuttonswrapper').css({ 'height': buttonsheight });
            } else {
                $('#menuhr').removeAttr('style');
                $('#actionbuttons').removeAttr('style');
                $('#actionbuttonswrapper').removeAttr('style');
            }
        };

        // run our function on load
        sticky_button_navigation();

        // and run it again every time you scroll
        $(window).scroll(function () {
            sticky_button_navigation();
        });
    }
