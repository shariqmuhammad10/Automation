    function SaveRegister() {

        var PayrollPeriodId = $("#PayrollPeriodId").val();
        var TransactionDate = $("#TransactionDate").val();
        var PayDate = $("#PayDate").val();
        var RegisterType = $("#RegisterType").val();

        var validationMessage = '';

        if (typeof PayrollPeriodId == 'undefined' || PayrollPeriodId == '')
            validationMessage += 'Payroll Period is required.' + '&lt;br /&gt;';

        if (typeof TransactionDate == 'undefined' || TransactionDate == '')
            validationMessage += 'Transaction Date is required.' + '&lt;br /&gt;';

        if (typeof PayDate == 'undefined' || PayDate == '')
            validationMessage += 'Pay Date is required.' + '&lt;br /&gt;';

        if (typeof RegisterType == 'undefined' || RegisterType == '')
            validationMessage += 'Register Type is required.' + '&lt;br /&gt;';

        if (validationMessage != '') {

            ShowAjaxMessage(validationMessage, "error");
            return;
        }

        var form = $("#FormPayrollRegister");

        $.ajax({
            type: "POST",
            async: false,
            cache: false,
            url: "/QAAutom/Payroll/Payroll/Save",
            data: form.serialize(),
            success: function (result) {
                if (result.Success === true) {
                    ShowAjaxMessage("", "");
                    ShowAjaxMessage(result.Message, "success");
                    if (result.NewRegister) {
                        window.location.href = "/QAAutom/Payroll/Payroll/RegisterDetails" + "?id=" + result.PayrollRegisterId + "&amp;addMode=" + result.NewRegister;
                    }
                    else {
                        window.location.href = "/QAAutom/Payroll/Payroll/RegisterDetails" + "?id=" + result.PayrollRegisterId;
                    }
                }
                else {
                    ShowAjaxMessage("", "");
                    ShowAjaxMessage(result.Message, "error");
                }
            },
            error: function (jqXHR, textStatus, errorThrown) {
                console.log(textStatus, errorThrown);
            }
        });
    }

    function doStickyButtons() {
        var sticky_button_offset_top = $('#menuhr').offset().top;

        // our function that decides weather the navigation bar should have "fixed" css position or not.
        var sticky_button_navigation = function () {
            var scroll_top = $(window).scrollTop(); // our current vertical position from the top
            var screenwidth = $(window).width();
            var buttonswidth = $('#actionbuttons').width();
            var buttonsheight = $('#actionbuttons').height();

            // if we've scrolled more than the navigation, change its position to fixed to stick to top,
            // otherwise change it back to relative
            if (scroll_top &gt; sticky_button_offset_top) {
                $('#menuhr').css({ 'position': 'fixed', 'top': 0, 'left': 40, 'z-index': 11, 'width': (screenwidth - 80) });
                $('#actionbuttons').css({ 'position': 'fixed', 'top': 10, 'right': 48, 'z-index': 12, 'width': buttonswidth });
                $('#actionbuttonswrapper').css({ 'height': buttonsheight });
            } else {
                $('#menuhr').removeAttr('style');
                $('#actionbuttons').removeAttr('style');
                $('#actionbuttonswrapper').removeAttr('style');
            }
        };

        // run our function on load
        sticky_button_navigation();

        // and run it again every time you scroll
        $(window).scroll(function () {
            sticky_button_navigation();
        });
    }
