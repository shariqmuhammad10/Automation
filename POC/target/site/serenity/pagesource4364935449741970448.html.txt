    function SaveRegister() {

        var PayrollPeriodId = $("#PayrollPeriodId").val();
        var TransactionDate = $("#TransactionDate").val();
        var PayDate = $("#PayDate").val();
        var RegisterType = $("#RegisterType").val();

        var validationMessage = '';

        if (typeof PayrollPeriodId == 'undefined' || PayrollPeriodId == '')
            validationMessage += 'Payroll Period is required.' + '&lt;br /&gt;';

        if (typeof TransactionDate == 'undefined' || TransactionDate == '')
            validationMessage += 'Transaction Date is required.' + '&lt;br /&gt;';

        if (typeof PayDate == 'undefined' || PayDate == '')
            validationMessage += 'Pay Date is required.' + '&lt;br /&gt;';

        if (typeof RegisterType == 'undefined' || RegisterType == '')
            validationMessage += 'Register Type is required.' + '&lt;br /&gt;';

        if (validationMessage != '') {

            ShowAjaxMessage(validationMessage, "error");
            return;
        }

        var form = $("#FormPayrollRegister");

        $.ajax({
            type: "POST",
            async: false,
            cache: false,
            url: "/QAAutom/Payroll/Payroll/Save",
            data: form.serialize(),
            success: function (result) {
                if (result.Success === true) {
                    ShowAjaxMessage("", "");
                    ShowAjaxMessage(result.Message, "success");
                    if (result.NewRegister) {
                        window.location.href = "/QAAutom/Payroll/Payroll/RegisterDetails" + "?id=" + result.PayrollRegisterId + "&amp;addMode=" + result.NewRegister;
                    }
                    else {
                        window.location.href = "/QAAutom/Payroll/Payroll/RegisterDetails" + "?id=" + result.PayrollRegisterId;
                    }
                }
                else {
                    ShowAjaxMessage("", "");
                    ShowAjaxMessage(result.Message, "error");
                }
            },
            error: function (jqXHR, textStatus, errorThrown) {
                console.log(textStatus, errorThrown);
            }
        });
    }

    function doStickyButtons() {
        var sticky_button_offset_top = $('#menuhr').offset().top;

        // our function that decides weather the navigation bar should have "fixed" css position or not.
        var sticky_button_navigation = function () {
            var scroll_top = $(window).scrollTop(); // our current vertical position from the top
            var screenwidth = $(window).width();
            var buttonswidth = $('#actionbuttons').width();
            var buttonsheight = $('#actionbuttons').height();

            // if we've scrolled more than the navigation, change its position to fixed to stick to top,
            // otherwise change it back to relative
            if (scroll_top &gt; sticky_button_offset_top) {
                $('#menuhr').css({ 'position': 'fixed', 'top': 0, 'left': 40, 'z-index': 11, 'width': (screenwidth - 80) });
                $('#actionbuttons').css({ 'position': 'fixed', 'top': 10, 'right': 48, 'z-index': 12, 'width': buttonswidth });
                $('#actionbuttonswrapper').css({ 'height': buttonsheight });
            } else {
                $('#menuhr').removeAttr('style');
                $('#actionbuttons').removeAttr('style');
                $('#actionbuttonswrapper').removeAttr('style');
            }
        };

        // run our function on load
        sticky_button_navigation();

        // and run it again every time you scroll
        $(window).scroll(function () {
            sticky_button_navigation();
        });
    }

    function AcMultipleAllocation_GetTotalAmount(unitsFN, unitPriceFN, amountFN, validate, precision) {
        var totalAmount = 0.0;

        if (amountFN != 'undefined' &amp;&amp; amountFN != '') {
            var amount = $("#" + amountFN).val();
            if (typeof amount != 'undefined') {
                if (validate &amp;&amp; (amount == '' || amount == '')) {
                    return null;
                }
                totalAmount = Number(amount);
            }
        }
        else {
            var orderQty = $("#" + unitsFN).val();
            var prUnitPrice = $("#" + unitPriceFN).val();
            if (typeof orderQty != 'undefined' &amp;&amp; (typeof prUnitPrice != 'undefined')) {
                if (validate &amp;&amp; (orderQty == '' || prUnitPrice == '' || orderQty == null || prUnitPrice == null)) {
                    return null;
                }
                totalAmount = Number(orderQty) * Number(prUnitPrice);
            }
        }

        return totalAmount.toFixed(precision);
    }

    function AcMultipleAllocation_GetTotalAmount(unitsFN, unitPriceFN, amountFN, validate, precision) {
        var totalAmount = 0.0;

        if (amountFN != 'undefined' &amp;&amp; amountFN != '') {
            var amount = $("#" + amountFN).val();
            if (typeof amount != 'undefined') {
                if (validate &amp;&amp; (amount == '' || amount == '')) {
                    return null;
                }
                totalAmount = Number(amount);
            }
        }
        else {
            var orderQty = $("#" + unitsFN).val();
            var prUnitPrice = $("#" + unitPriceFN).val();
            if (typeof orderQty != 'undefined' &amp;&amp; (typeof prUnitPrice != 'undefined')) {
                if (validate &amp;&amp; (orderQty == '' || prUnitPrice == '' || orderQty == null || prUnitPrice == null)) {
                    return null;
                }
                totalAmount = Number(orderQty) * Number(prUnitPrice);
            }
        }

        return totalAmount.toFixed(precision);
    }

            function getGrossPayonFly() {
                if (getGrossPayonFlyInProgress)
                    return;
                ShowAjaxMessage("", "", "PayrollRegisterJobEdit_Error");

                getGrossPayonFlyInProgress = true;

                var id = $("#RecordId").val();
                var regularHours = $("#RegularHours").val();
                var ot = $("#OT").val();
                var dockHours = $('input[name="DockHours"]').val();
                var dockAmount = $('input[name="DockAmount"]').val();
                var equivalentOTHours = PayrollRegisterJobEditGetEquivalentOTHours(ot);
                var overtimeHourlyRateUsed = $("#PayrollRegisterJobDetail_OvertimeHourlyRateUsed").val();

                var data = { id: id, regularHours: regularHours, ot: ot, dockHours: dockHours, dockAmount: dockAmount, exceedsDockAmount: false, equivalentOTHours: equivalentOTHours, overtimeHourlyRateUsed: overtimeHourlyRateUsed };

                $.ajax({
                    cache: false,
                    url: "/QAAutom/Payroll/Payroll/GetGrossPay",
                    data: data,
                    dataType: 'JSON',
                    success: function (result) {
                        if (result != null) {
                            if (result.Success) {
                                if ($('#GrossPay') &amp;&amp; $('#GrossPay').data("kendoNumericTextBox")) {
                                    $('#GrossPay').data("kendoNumericTextBox").value(result.GrossPay);
                                    $('#GrossPaySalary').data("kendoNumericTextBox").value(result.GrossPay);
                                    $('input[name="DockAmount"]').val(result.DockAmount);
                                    getGrossPayonFlyInProgress = false;
                                }
                            }
                            else if (result.Message.search('entered for this job is greater than') &gt;= 0) {
                                ShowYesNoDialog('Selection', result.Message, function () { getGrossPayonFlyExceedsDockAmount(data); }, function () { getGrossPayonFlyInProgress = false; });
                            }
                            else {
                                ShowAjaxMessage(result.Message, "error", "PayrollRegisterJobEdit_Error");
                                getGrossPayonFlyInProgress = false;
                            }
                        }
                    }
                });
            };

            };

            function PayrollRegisterJobEditGetEquivalentOTHours(ot) {
                var equivalentOTHours = ot * 1.5;
                var len = registerJobEarningOvertimeItemsCache.length;
                if (len == 1) {
                    if (registerJobEarningOvertimeItemsCache[0].OvertimeHoursMultiplier != null) {
                        equivalentOTHours = ot * registerJobEarningOvertimeItemsCache[0].OvertimeHoursMultiplier;
                    }
                }
                else if (len &gt; 1) {
                    equivalentOTHours = 0;
                    for (var i = 0; i &lt; len; i++) {
                        if (registerJobEarningOvertimeItemsCache[i].OvertimeHoursMultiplier == null) {
                            equivalentOTHours += registerJobEarningOvertimeItemsCache[i].ActualOvertimeHours * 1.5;
                        }
                        else {
                            equivalentOTHours += registerJobEarningOvertimeItemsCache[i].ActualOvertimeHours * registerJobEarningOvertimeItemsCache[i].OvertimeHoursMultiplier;
                        }
                    }
                }
                return equivalentOTHours;
            }

            function CalculateOvertimeRatePerHour(otRate, otMultiplier) {
                otRate = Math.round(Number(otRate) * 100) / 100;
                if (otMultiplier == null) {
                    otMultiplier = 1.5;
                }
                else {
                    otMultiplier = Math.round(Number(otMultiplier) * 10000) / 10000;
                }
                return Math.round(otRate * otMultiplier * 1000) / 1000;
            }

            function CalculateEquivalentOvertimeHours(ot, otMultiplier) {
                ot = Math.round(Number(ot) * 10000) / 10000;
                if (otMultiplier == null) {
                    otMultiplier = 1.5;
                }
                else {
                    otMultiplier = Math.round(Number(otMultiplier) * 10000) / 10000;
                }
                return Math.round(ot * otMultiplier * 10000) / 10000;
            }

            function CalculateOvertimeItemAmount(ot, otRate, otMultiplier) {
                ot = Math.round(Number(ot) * 10000) / 10000;
                otRate = Math.round(Number(otRate) * 1000) / 1000;
                if (otMultiplier == null) {
                    otMultiplier = 1.5;
                }
                else {
                    otMultiplier = Math.round(Number(otMultiplier) * 10000) / 10000;
                }
                return Math.round(ot * otRate * otMultiplier * 100) / 100;
            }

            function SaveRegisterOTItemAllocation(earningOTItems, overtimeSequence, otHours, otMultiplier, otAmount, accountCode) {
                var len = earningOTItems.length;
                for (var i = 0; i &lt; len; i++) {
                    if (earningOTItems[i].OvertimeSequence == overtimeSequence) {
                        earningOTItems[i].ActualOvertimeHours = otHours;
                        earningOTItems[i].OvertimeHoursMultiplier = otMultiplier;
                        earningOTItems[i].EquivalentOvertimeHours = CalculateEquivalentOvertimeHours(otHours, otMultiplier);
                        earningOTItems[i].OvertimeItemAmount = otAmount;
                        earningOTItems[i].AccountCode = accountCode;
                        break;
                    }
                }
            }
