    function RefreshOnSubmitMassRequisition(requisitionId, overbudget, moveNext, key) {
        var operationResultId = GetMassActionId();

        $("#divPRMyRequisitionDataView").data("AptaDataView").ExecuteMassOperation('/QAAutom/Purchasing/MyPurchaseRequisition/SubmitMassRequsition',
                     { requisitionId: requisitionId, overBudget: overbudget, moveNext: moveNext, operationResultId: operationResultId, key: key },
                key, function (result) {
                    if (result.Success == 1) {
                        RefreshOnSubmitMassRequisition(result.InvoiceId, false, false, key);

                    } else if (result.Success == 3) {

                        if (submitForAllYes === true) {
                            RefreshOnSubmitMassRequisition(result.InvoiceId, true, false, key);
                        }
                        else if (submitForAllNo === true) {
                            RefreshOnSubmitMassRequisition(result.InvoiceId, false, true, key);
                        }
                        else {

                            if ('False' == 'True') {
                                result.Message += "&lt;br /&gt;&lt;br /&gt;Do you wish to continue?"
                                ShowYesNoDialogWithCheckBox("My Purchase Requisitions", result.Message, function (e) { RefreshOnSubmitMassRequisition(result.InvoiceId, true, false, key); submitForAllYes = e; }, function (e) { RefreshOnSubmitMassRequisition(result.InvoiceId, false, true, key); submitForAllNo = e; });
                            }
                            else {
                                // create an appropriate error message
                                result.Message += "&lt;br /&gt;&lt;br /&gt;Purchase Request cannot be submitted."
                                ShowOkDialog("My Purchase Requisitions", result.Message, null);
                            }
                        }
                    }
                    else if (result.Success == 4) {

                        //4 indicates process end,
                        submitForAllNo = false;
                        submitForAllYes = false;
                        var grid = $("#PRMyRequisitionDataView").data("kendoGrid");
                        ClearGridSelectionAndReload(grid, "/QAAutom/GridConfig/SetGridSelectedRecords");
                        ShowResultGrid();
                    }
                    else {
                        RefreshOnSubmitMassRequisition(result.InvoiceId, false, false, key);

                    }
                }, "popupwindowMassActionResults", false);
    }

    function doStickyButtons() {
        var sticky_button_offset_top = $('#menuhr').offset().top;

        // our function that decides weather the navigation bar should have "fixed" css position or not.
        var sticky_button_navigation = function () {
            var scroll_top = $(window).scrollTop(); // our current vertical position from the top
            var screenwidth = $(window).width();
            var buttonswidth = $('#actionbuttons').width();
            var buttonsheight = $('#actionbuttons').height();

            // if we've scrolled more than the navigation, change its position to fixed to stick to top,
            // otherwise change it back to relative
            if (scroll_top &gt; sticky_button_offset_top) {
                $('#menuhr').css({ 'position': 'fixed', 'top': 0, 'left': 40, 'z-index': 11, 'width': (screenwidth - 80) });
                $('#actionbuttons').css({ 'position': 'fixed', 'top': 10, 'right': 48, 'z-index': 12, 'width': buttonswidth });

            } else {
                $('#menuhr').removeAttr('style');
                $('#actionbuttons').removeAttr('style');
                $('#actionbuttonswrapper').removeAttr('style');
            }
        };

        // run our function on load
        sticky_button_navigation();

        // and run it again every time you scroll
        $(window).scroll(function () {
            sticky_button_navigation();
        });
    }

    $('#btnSaveAndSubmit__PurchaseRequisitionView').click(function (e) {

    function SaveBeforeAnyOperation(e, operation) {
        $("#PROperation").val(0);
        $("#PRIsSubmit").val(true);
        ShowAjaxMessage("", "", "PurchaseRequisitionDetailPopup_Error");
        var destinationID = $("#DestinationId").val();

        var rName;
        var requisitionerId;
        var validationMessage = '';

        if ('0' &gt; 0 &amp;&amp; ('My Purchase Requisitions' == 'Admin Purchase Requisitions' || 'My Purchase Requisitions' == 'My Purchase Requisition Approvals')) {
            rName = $('#RequestorName').val();
            requisitionerId = '';
            $("#RequestorName").data("FINSUITextInput").SetValue(rName);
            $('#RequisitionerId').val(requisitionerId);
        }
        else {
            rName = $('#RequisitionerId').data("kendoComboBox").text();
            requisitionerId = $('#RequisitionerId').data("kendoComboBox").value();
            $("#RequisitionerId").data("FINSComboBox").SetValue(requisitionerId);
            $('#RequestorName').val(rName);
        }

        if (typeof destinationID == 'undefined' || destinationID == '')
            validationMessage += 'Ship To is required' + '&lt;br /&gt;';
        if (typeof requisitionerId == 'undefined' || requisitionerId == '')
            validationMessage += 'Requisitioner Name is required.' + '&lt;br /&gt;';

        if (validationMessage != '') {

            ShowAjaxMessage(validationMessage, "error", "PurchaseRequisitionDetailPopup_Error");
            return;
        }

        var form = $("#RequisitionDetails");
        $.ajax({
            type: "POST",
            async: false,
            url: form.attr('action'),
            data: form.serialize(),
            error: function (xhr, status, error) {
            },
            success: function (result) {
                if (result.Success == true) {

                    if (operation == "Submit")
                    {
                        SubmitRequisitions(false, e);
                    }

                    else if (operation == "Approve")
                    {
                        ApproveRequisitionOnPopup(e);
                    }

                    else if (operation == "Disapprove")
                    {
                        RefreshPopupAfterDisapproveRequisition(e);
                    }

                    else if (operation == "CopyRequisition")
                    {
                        CopyRequisitionPopup(e);
                    }

                }
                else {
                    ShowAjaxMessage(result.Message, "error", "PurchaseRequisitionDetailPopup_Error");
                }

            }
        });
    }

    function SubmitRequisitions(overBudget, e) {
        ShowAjaxMessage("", "");
        ShowAjaxMessage("", "", "PurchaseRequisitionDetailPopup_Error");
        var data = { overBudget: overBudget, requisitionId: '0' };
        $.ajax({
            type: "POST",
            async: false,
            url: "/QAAutom/Purchasing/MyPurchaseRequisition/SubmitRequisition",
            data: data,
            success: function (result) {

                if (result.success == true) {
                    var navigationID = 0;

                    if(!$('#btnSaveAndNext__PurchaseRequisitionView').is(":visible")){
                        navigationID = '0';
                    }
                    else{
                        navigationID = '';
                    }
                    var aptaDataViewSelection;
                    var grid = $("#"+'PRMyRequisitionDataView').data("kendoGrid");
                    if(grid != null)
                    {
                        aptaDataViewSelection = $("#div"+'PRMyRequisitionDataView').data("AptaDataView").GetAptaDataViewSelection();
                    }
                    else{
                        aptaDataViewSelection = null;
                    }

                    $(e.target).closest(".k-window-content").data("FINSWindow").UpdateContent({url:"/QAAutom/Purchasing/MyPurchaseRequisition/EditMyRequisitionPopup",data: {id : navigationID, origin:'MY_PURCHASE_REQUISITION' , aptaDataViewSelection: aptaDataViewSelection, statusType: '3', adminPRApproverId: '' },type:"POST"});
                    ShowAjaxMessage(result.Message, "success","PurchaseRequisitionDetailPopup_Error");
                }
                else if (result.success == false) {
                    if ('False' == 'True') {
                        result.Message += "&lt;br /&gt;&lt;br /&gt;Do you wish to continue?"
                        ShowYesNoDialog("Purchase Requisition", result.Message, function () { SubmitRequisitions(true, e); });
                    }
                    else {
                        result.Message += "&lt;br /&gt;&lt;br /&gt;Purchase Requisition cannot be submitted."
                        ShowOkDialog("Purchase Requisition", result.Message, null);
                    }
                }
                else {
                    ShowAjaxMessage(result, "error", "PurchaseRequisitionDetailPopup_Error");
                }
            },
            error: function (jqXHR, textStatus, errorThrown) {
                console.log(textStatus, errorThrown);
            }
        });
    }
