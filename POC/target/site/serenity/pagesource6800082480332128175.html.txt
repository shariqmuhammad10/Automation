    {
        var finsWindow = FINSWindow.FinsWindowManager.CreateMassActionWindowObject({ url: "/QAAutom/AccountsPayable/VendorInvoice/MassActionResults" }).SetonRefresh(function () { RefreshOnSubmitForPayment(-1, false, false, 'VendorInvoiceDataView', false, false); });

    var finspopupwindowCloseInvoicePO = null;

    function RefreshOnSubmitForPayment(invoiceId, overbudget, moveNext, key, associatedOrderReceipt, operationCancelled) {
        var operationResultId = GetMassActionId();
        finspopupwindowCloseInvoicePO = FINSWindow.FinsWindowManager.CreateFINSWindowObject("popupwindowCloseInvoicePO")
                                                    .SetContent({ url:"/QAAutom/AccountsPayable/VendorInvoice/ClosePO"}).SetIsModal(true)
                                        .SetTitle("Close Dollar PO").SetonClose(function () { ShowResultGrid(); }).SetonRefresh(function () {
                                            $("#divVendorInvoiceDataView").data("AptaDataView").ExecuteMassOperation('/QAAutom/AccountsPayable/VendorInvoice/SubmitForPayment',
                         { invoiceId: invoiceId, overBudget: overbudget, moveNext: moveNext, operationResultId: operationResultId, key: key, associatedOrderReceipt : associatedOrderReceipt, operationCancelled : operationCancelled },
                    key, function (result) {
                        if(submitForAllYesOR)
                        {
                            associatedOrderReceipt = true;
                        }
                        else
                        {
                            associatedOrderReceipt = false;
                        }

                        if (result.Success == 1) {
                            RefreshOnSubmitForPayment(result.InvoiceId, false, false, key, associatedOrderReceipt, false);

                        } else if (result.Success == 3) {

                            if (submitForAllYes === true) {
                                RefreshOnSubmitForPayment(result.InvoiceId, true, false, key, true, false);
                            }
                            else if (submitForAllNo === true) {
                                RefreshOnSubmitForPayment(result.InvoiceId, false, true, key, true, false);
                            }
                            else {
                                // Fix grammar error coming from database
                                result.Message = result.Message.replace("The invoice of", "The invoice");

                                if (1 == true)
                                {
                                    ShowYesNoDialogWithCheckBox("Vendor Invoice", result.Message, function (e) { RefreshOnSubmitForPayment(result.InvoiceId, true, false, key, true, false); submitForAllYes = e; }, function (e) { RefreshOnSubmitForPayment(result.InvoiceId, false, true, key, true, false); submitForAllNo = e; });
                                }
                                else
                                {
                                    // create an appropriate error message
                                    var intStartIndex = result.Message.indexOf("Do you wish to submit for payment anyway?");
                                    var strOverBudgetMessage = result.Message.substring(0, intStartIndex-1);
                                    strOverBudgetMessage += "&amp;nbsp;&amp;nbsp;This Invoice cannot be submitted for payment.";
                                    ShowOkDialog("Vendor Invoice", strOverBudgetMessage, null);
                                }
                            }
                        }

                        else if (result.Success == 5) {
                            if (submitForAllYesOR === true) {
                                RefreshOnSubmitForPayment(result.InvoiceId, false, false, key, true, false);
                            }
                            else if (submitForAllNoOR === true) {
                                RefreshOnSubmitForPayment(result.InvoiceId, false, true, key, false, true);
                            }
                            else {
                                ShowYesNoDialogWithCheckBox("Vendor Invoice", "This invoice does not have a related Order Receipt for the items being invoiced. This invoice will be placed in the Pending state until an Order Receipt is created and submitted. You can submit this invoice a second time, which will create a Pending Order Receipt in the Receiving module.", function (e) { RefreshOnSubmitForPayment(result.InvoiceId, false, false, key, true, false); submitForAllYesOR = e; }, function (e) { RefreshOnSubmitForPayment(result.InvoiceId, false, true, key, false, true); submitForAllNoOR = e; });
                            }
                        }
                        else if (result.Success == 4) {

                            //4 indicates process end,
                            submitForAllNo = false;
                            submitForAllYes = false;
                            submitForAllNoOR = false;
                            submitForAllYesOR = false;
                            var grid = $("#VendorInvoiceDataView").data("kendoGrid");
                            ClearGridSelectionAndReload(grid, "/QAAutom/GridConfig/SetGridSelectedRecords");
                            var VendorInvoiceCloseGrid = $("#VendorInvoiceClosePODataView").data("kendoGrid");

                            VendorInvoiceCloseGrid.dataSource.bind("change", function(){
                                if(VendorInvoiceCloseGrid.dataSource.data().length &gt; 0){
                                    if(finspopupwindowCloseInvoicePO != null)
                                        finspopupwindowCloseInvoicePO.OpenWindow();
                                }
                                else{
                                    ShowResultGrid();
                                }

                            });
                            VendorInvoiceCloseGrid.dataSource.read();

                        }
                        else {
                            RefreshOnSubmitForPayment(result.InvoiceId, true, false, key, true, false);

                        }
                    }, "popupwindowMassActionResults", false);

                                        });
        finspopupwindowCloseInvoicePO.CreateWindow(false);

    }

    function doStickyButtons() {
        var sticky_button_offset_top = $('#menuhr').offset().top;

        // our function that decides weather the navigation bar should have "fixed" css position or not.
        var sticky_button_navigation = function () {
            var scroll_top = $(window).scrollTop(); // our current vertical position from the top
            var screenwidth = $(window).width();
            var buttonswidth = $('#actionbuttons').width();
            var buttonsheight = $('#actionbuttons').height();

            // if we've scrolled more than the navigation, change its position to fixed to stick to top,
            // otherwise change it back to relative
            if (scroll_top &gt; sticky_button_offset_top) {
                $('#menuhr').css({ 'position': 'fixed', 'top': 0, 'left': 40, 'z-index': 11, 'width': (screenwidth - 80) });
                $('#actionbuttons').css({ 'position': 'fixed', 'top': 10, 'right': 48, 'z-index': 12, 'width': buttonswidth });
                $('#actionbuttonswrapper').css({ 'height': buttonsheight });
            } else {
                $('#menuhr').removeAttr('style');
                $('#actionbuttons').removeAttr('style');
                $('#actionbuttonswrapper').removeAttr('style');
            }
        };

        // run our function on load
        sticky_button_navigation();

        // and run it again every time you scroll
        $(window).scroll(function () {
            sticky_button_navigation();
        });
    }
