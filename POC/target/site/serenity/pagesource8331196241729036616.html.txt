    function ApproveDisapprove(approveType) {

        //var appIds = approvers.split('~');
        var approverId = '';

        var grid = $("#PRApprovalsDataView").data("kendoGrid");

        var isChecked = ($('#chkRequireMyAttention_ApproverList').is(":checked"));

        var arr = selectedRowIds.split('~');

            var data = { approverId: approverId, approvalTrackingId: selectedRowIds, approvalType: '3' };

            $.ajax({
                url: '/QAAutom/Purchasing/RequisitionApproval/RequisitionApproveAction',

                dataType: 'JSON',
                data: data,
                success: function (result) {
                    var grid = $("#PRApprovalsDataView").data("kendoGrid");
                    if (result.Success == 1) {
                        ShowAjaxMessage(result.Message, "success");
                    }
                    else if (result.Success == 2) {
                        ShowAjaxMessage(result.Message, "notice");
                    }
                    else {
                        ShowAjaxMessage(result.Message, "error");
                    }
                    ClearGridSelectionAndReload(grid, "/QAAutom/GridConfig/SetGridSelectedRecords");
                },
                error: function (jqXHR, textStatus, errorThrown) {
                    console.log(textStatus, errorThrown);
                }

            if (arr[1] != 'Submitted') {
                ShowAjaxMessage("Cannot disapprove Requisition, not in Sumbit status", "error");
                return;

            data = { approverId: approverId, requireMyAttention: isChecked, approvalTrackingId: selectedRowIds, approvalType: '3', Ismultiple: true };

        var sticky_button_navigation = function () {
